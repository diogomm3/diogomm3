<!-- Script to update year progress dynamically -->
<script>
  function updateYearProgress() {
    const now = new Date();
    const start = new Date(now.getFullYear(), 0, 1);
    const diff = now - start;
    const oneYear = 31556952000; // milliseconds in a year
    const progress = Math.min((diff / oneYear) * 100, 100);
    const percent = Math.round(progress);
    
    const barLength = 10;
    const filledLength = Math.min(Math.floor((barLength * progress) / 100), barLength);
    const emptyLength = barLength - filledLength;
    
    const bar = '█'.repeat(filledLength) + '░'.repeat(emptyLength);
    
    document.getElementById('progress-bar').textContent = bar;
    document.getElementById('progress-percent').textContent = percent + '%';
  }
  
  // Update on load and every minute
  window.addEventListener('load', updateYearProgress);
  setInterval(updateYearProgress, 60000);
</script>
